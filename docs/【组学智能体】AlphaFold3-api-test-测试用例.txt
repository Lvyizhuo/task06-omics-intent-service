3、alphafold3官方测试命令（单次任务）
docker run --gpus '"device=1"' \
    -v /data1t/ntt/lvyizhuo/alphafold3/weights:/root/models \
    -v /data1t/ntt/lvyizhuo/alphafold3/databases:/root/public_databases \
    -v /data2/ntt/lvyizhuo/task06-alphafold3-agent/input:/root/input \
    -v /data2/ntt/lvyizhuo/task06-alphafold3-agent/output:/root/output \
    alphafold3 \
    uv run python3 run_alphafold.py \
        --json_path=/root/input/test.json \
        --model_dir=/root/models \
        --output_dir=/root/output



3、服务启动（docker compose 方式，推荐）：
# 在项目目录下执行
docker compose up -d

# 或使用原始 docker run 方式：
docker run -d \
    --gpus '"device=1"' \
    --name alphafold3-api \
    --restart unless-stopped \
    -p 8015:8015 \
    -v /data1t/ntt/lvyizhuo/alphafold3/weights:/root/models:ro \
    -v /data1t/ntt/lvyizhuo/alphafold3/databases:/root/public_databases:ro \
    -v /data1t/ntt/lvyizhuo/task06-alphafold3-agent/storage:/app/storage \
    -v /data1t/ntt/lvyizhuo/task06-alphafold3-agent/data:/app/data \
    -v /data1t/ntt/lvyizhuo/task06-alphafold3-agent/logs:/app/logs \
    alphafold3-api
